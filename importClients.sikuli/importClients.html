from sikuli import *
import logging
import myTools

# - - - - - - - - - - - - - - - - - - - - - - - - - #
def Add_CustomField(name, downArrow):
# - - - - - - - - - - - - - - - - - - - - - - - - - #

    type(Key.ENTER)
    type(Key.TAB)
    type("a",KEY_CTRL)

    for i in range(downArrow+1):
        type(Key.DOWN)

    type(Key.ENTER)
    type(Key.TAB,KEY_SHIFT)
