def baLogHeader():
# - - - - - - - - - - - - - - - - - - - - - - - - - #

    if os.path.isfile(Settings.BALogFile):
        outFile = Settings.repFolder + "\\BA-Log.txt"
        billLog = open(outFile, "a")
        billLog.write("\n")
        billLog.write("==================================================\n")
        baTime = datetime.datetime.now()
        billLog.write(baTime.strftime("Started: %Y-%m-%d %H:%M:%S\n"))
        billLog.write("==================================================\n")
